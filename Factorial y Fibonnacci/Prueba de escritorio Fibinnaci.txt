fib(0, 1) :- !.
fib(1, 1) :- !.
fib(N, F) :-
    N >= 1,
    N1 is N - 1,
    N2 is N - 2,
    fib(N1, F1),
    fib(N2, F2),
    F is F1 + F2.

fib(4, F):
    4 > 1
    N1 = 3
    N2 = 2
    fib(3, F1) --> F1 = 3
    fib(2, F2) --> F2 = 2
    F is F1 + F2 --> F = 5

fib(3, F1):
   3 > 1
   N1 = 2
   N2 = 1
   fib(2, F3) --> F3 = 2
   fib(1, F4) --> F4 = 1
   F1 is F3 + F4 --> F1 = 3

fib(2, F3):
    2 > 1
    N1 = 1
    N2 = 0
    fib(1, 1) --> 1
    fib(0, 1) --> 1
    F3 is 1 + 1 --> F3 = 2
